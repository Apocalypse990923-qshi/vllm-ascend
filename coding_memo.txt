int n = nums1.size();
vector<int> segtree(4*n+1);  //规模一般设置为4*n+1
vector<bool> tag(4*n+1);	//懒惰标记
build(segtree,0,nums1,0,n-1);

void build(vector<int>& segtree, int index, vector<int>& arr, int st, int ed){
        if(st==ed){
            segtree[index] = arr[st];
            return;
        }
        int m = (st+ed)/2;
        build(segtree,index*2+1,arr,st,m);
        build(segtree,index*2+2,arr,m+1,ed);
        segtree[index] = segtree[index*2+1] + segtree[index*2+2];
    }

    int query(vector<int>& segtree, int index, int l, int r, int st, int ed, vector<bool>& tag){
        if(st<=l && r<=ed){
            return segtree[index];
        }
        if(r<st || ed<l)    return 0;
        int ret = 0;
        int m = (l+r)/2;
        if(tag[index]){
            tag[index]=false;
            tag[2*index+1]= !tag[2*index+1];
            tag[2*index+2]= !tag[2*index+2];
            segtree[2*index+1] = (m-l+1)-segtree[2*index+1];
            segtree[2*index+2] = (r-(m+1)+1)-segtree[2*index+2];
        }
        if(st<=m){
            ret+=query(segtree, index*2+1, l, m, st, ed, tag);
        }
        if(ed>=m+1){
            ret+=query(segtree,index*2+2, m+1, r, st, ed, tag);
        }
        return ret;
    }

    void modify(vector<int>& segtree, int index, int l, int r, int st, int ed, vector<bool>& tag){
        if(st<=l && r<=ed){
            segtree[index] = r-l+1-segtree[index];
            tag[index] = !tag[index];
            return;
        }
        if(r<st || ed<l)    return;
        int m = (l+r)/2;
        if(tag[index]){
            tag[index]=false;
            tag[2*index+1]= !tag[2*index+1];
            tag[2*index+2]= !tag[2*index+2];
            segtree[2*index+1] = (m-l+1)-segtree[2*index+1];
            segtree[2*index+2] = (r-(m+1)+1)-segtree[2*index+2];
        }
        if(st<=m){
            modify(segtree, index*2+1, l, m, st, ed, tag);
        }
        if(ed>=m+1){
            modify(segtree,index*2+2, m+1, r, st, ed, tag);
        }
        segtree[index] = segtree[2*index+1] + segtree[2*index+2];
    }
